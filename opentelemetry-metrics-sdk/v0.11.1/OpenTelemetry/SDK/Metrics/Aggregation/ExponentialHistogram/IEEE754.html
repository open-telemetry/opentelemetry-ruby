def self.get_ieee_754_exponent(value)
  bits = [value].pack('d').unpack1('Q')
  ((bits & EXPONENT_MASK) >> MANTISSA_WIDTH) - EXPONENT_BIAS
end

def self.get_ieee_754_mantissa(value)
