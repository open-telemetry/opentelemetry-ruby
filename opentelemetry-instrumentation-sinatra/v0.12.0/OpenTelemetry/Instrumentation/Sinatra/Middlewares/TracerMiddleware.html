def call(env)
  tracer.in_span(
    env['PATH_INFO'],
    attributes: { 'http.method' => env['REQUEST_METHOD'],
                  'http.url' => env['PATH_INFO'] },
    kind: :server,
    with_parent: parent_context(env)
  ) do |span|
    app.call(env).tap { |resp| trace_response(span, env, resp) }
  end
end
